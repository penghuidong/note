```c++
#include <iostream>
#include <bitset>

using namespace std;

int main()
{
	unsigned int dw_bit_set = 0;
	bool status = false;

	/*
	 *  左移右移优先级7
	 *  &，^, |优先级依次是10,11,12
	 *  而 ~ 优先级最大，为3
	 */
	dw_bit_set |= 1UL << 27;  
	std::bitset<32> num_bit(dw_bit_set);
	std::cout << num_bit << std::endl;
	status = dw_bit_set & (1UL << 27);
	std::cout << "status = " << status << std::endl;

	dw_bit_set &= ~(1UL << 27); // ~(1UL << 27) -> 除第27位外都是1
	num_bit = dw_bit_set;
	std::cout << num_bit << std::endl;
	status = dw_bit_set & (1UL << 27);
	std::cout << "status = " << status << std::endl;

	return 0;
}
```
